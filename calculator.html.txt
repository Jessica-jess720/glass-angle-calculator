    <div class="calculator">
        <h2>弹道参数计算器</h2>
        <p>输入 L/S 比值（X值）：</p>
        <input type="number" step="any" id="inputX" placeholder="请输入X值">
        <button onclick="calculate()">计算</button>
        
        <div id="results">
            <h3>计算结果：</h3>
            <p>Tanθ (Y) = <span id="resultY">等待计算...</span></p>
            <p>θ角度 = <span id="resultAngle">等待计算...</span></p>
        </div>

        <div class="formula">
            <h4>使用公式：</h4>
            <p>Y = 2.317X - 2.245</p>
            <p>θ = arctan(Y)</p>
        </div>
    </div>

    <script>
        function calculate() {
            // 获取输入值
            let X = parseFloat(document.getElementById('inputX').value);
            
            // 计算Y值
            let Y = 2.317 * X - 2.245;
            
            // 计算角度（弧度转角度）
            let angleRad = Math.atan(Y);
            let angleDeg = angleRad * (180 / Math.PI);
            
            // 显示结果（保留4位小数）
            document.getElementById('resultY').textContent = Y.toFixed(4);
            document.getElementById('resultAngle').textContent = angleDeg.toFixed(2) + "°";
        }
    </script>
